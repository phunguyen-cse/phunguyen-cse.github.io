//Student.java
public class Student
{
	int grade;
	String name;

	// constructor
	public Student()
	{
	}

	public Student(String name, int grade)
	{
		this.grade = grade;
		this.name = name;
	}

	// accessor
	public int getGrade() {return grade;}
	public String getName() {return name;}

	public String getLetterGrade()
	{
		if (grade >= 90)
			return "A";
		else if (grade >= 80)
			return "B";
		else if (grade >= 70)
			return "C";
		else if (grade >= 60)
			return "D";
		else
			return "F";
	}

	// mutator
	public void setGrade(int grade)
	{
		this.grade = grade;
	}

	public void setName(String name)
	{
		this.name = name;
	}
}
			
//Switch.java
import java.util.Scanner;

public class Switch
{
	public static void main(String args[])
	{
		int grade;
		Scanner s = new Scanner(System.in);

		System.out.println("Enter your final grade:");
		grade = s.nextInt();

		Student st = new Student("David", grade);

		switch(st.getLetterGrade())
		{
			case "A":
				System.out.println("Final grade: A");
				break;
			case "B":
				System.out.println("Final grade: B");
				break;
			default:
				System.out.println("Work harder");
				break;
		}
	}
}
