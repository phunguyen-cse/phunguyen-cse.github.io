Adding Integers
/* Read two integer numbers from keyboard and calculate the summation
   01.24.2017
*/

import java.util.Scanner;//load class Scanner defined in JDK

public class Addition
{
	public static void main(String args[])
	{
		// create a Scanner object r
		Scanner r = new Scanner(System.in);

		// define three integer variables
		int num1, num2, sum;

		// read the first number
		System.out.println("Enter first number: ");
		num1 = r.nextInt();

		// read the second number
		System.out.println("Enter second number: ");
		num2 = r.nextInt();

		// calculate the summation of num1 and num2, then save results to sum
		sum = num1 + num2;

		// print out sum
		System.out.printf("Sum is: %d\n", sum);//%d is a placeholder for integer number
	}
}

  • import java.util.Scanner
    • import, load the pre-defined Java class library
    • Scanner, a class to read data
    • r, a variable whose data type is Scanner
    • new, create an object
    • System.in, standard input object, keyboard
    • nextInt(), read an integer number which is defined in Scanner class
    • =, assignment, assign the value on the right to the variable on the left side
  • Variable Name
    • Do not use reserved words, such as int, float
    • First character is a letter or "_", the rest of characters can be letter, digit, or "_"
    • Do not use symbols or spacce, such as "%", "^"
  • Data Type
    • int, -1, 0, 5, 10
    • float/double, 1.1, 3.1415926, -10.0
    • char, a, b, c, d

    import java.util.Scanner;

public class Relational
{
	public static void main(String args[])
	{
		// define two integer variables
		int num1, num2;
		Scanner s = new Scanner(System.in);

		// read the first number
		System.out.println("Enter first number: ");
		num1 = s.nextInt();

		// read the second number
		System.out.println("Enter second number: ");
		num2 = s.nextInt();

		// compare two numbers
		if (num1 > num2)
			System.out.printf("%d > %d\n", num1, num2);
		else
			System.out.printf("%d <= %d\n", num1, num2);
	}
}
