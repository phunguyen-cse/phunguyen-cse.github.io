For loop
public class Num
{
	public static void main(String args[])
	{
		for(int i = 0; i < 10; i++)
		{
			System.out.println(i);
		}
	}
}
			

public class Sum
{
	public static void main(String args[])
	{
		int sum = 0;

		for(int i = 0; i < 10; i++)
		{
			sum += i;
		}

		System.out.printf("Sum : %d\n", sum);
	}
}
Break and continue
public class Break
{
	public static void main(String args[])
	{
		for(int i = 0; i < 10; i++)
		{
			if (i == 6)
				break;

			System.out.println(i);
		}
	}
}
			
public class Continue
{
	public static void main(String args[])
	{
		for(int i = 0; i < 10; i++)
		{
			if (i == 6)
				continue;

			System.out.println(i);
		}
	}
}
Logical Operators

import java.util.Scanner;

public class Num2
{
	public static void main(String args[])
	{
		Scanner s = new Scanner(System.in);
		int num;

		num = s.nextInt();

		// AND
		if ((num > 10) && (num % 2 == 0))
			System.out.println("Greater than 10 and Even");

		// OR
		if ((num < 10) || (num > 20))
			System.out.println("Greater than 20 or Less than 10");

		// Not
		if (!(num % 2 == 0))
			System.out.println("Odd");
	}
}
Nested for loop
public class Nested
{
	public static void main(String args[])
	{
		for(int i = 0; i < 3; i++)
		{
			for(int j = 0; j < 3; j++)
			{
				System.out.printf("%d + %d = %d\n", i, j, i+j);
			}
		}
	}
}
Do ... while ...
public class While
{
	public static void main(String args[])
	{
		int i = 0;

		do
		{
			System.out.println(i);
			i++;
		}while (i < 10);
	}
}
