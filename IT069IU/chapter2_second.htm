Escape
/* Escape
 * CSC 230
 */

public class Escape
{
	public static void main(String args[])
	{
		//escape
		//\n, new line
		//\t, horizontal tab
		//\\, backslash
		//\", double quote
		System.out.println("Line 1: Welcome to\nLine 2: Java Programming!");
		System.out.println("\"Name:\" Lin");
	}

}
Printf
/* Printf
 * CSC 230
 */

public class Printf
{
	public static void main(String args[])
	{
		//%s is a placeholder for a string
		System.out.printf("%s -- %s\n%s!\n", "String 1", "String 2", "String 3");
		
		//set width
		System.out.printf("%20s -- %20s\n%20s!\n", "String 1", "String 2", "String 3");
	}

}
