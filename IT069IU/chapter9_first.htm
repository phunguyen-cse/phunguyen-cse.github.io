Student and CSStudent
public class Student
{
	private String name;

	public Student()
	{
	}

	public Student(String n)
	{
		name = n;
	}

	public String getName()
	{
		return name;
	}

	public void setName(String n)
	{
		name = n;
	}

	public static void main(String args[])
	{
		Student s = new Student("David");
		System.out.printf("%s\n", s.getName());
	}
}
			
public class CSStudent
{
	public String name;
	public String concentration;//either DS or IT

	public CSStudent()
	{
	}

	public CSStudent(String n, String c)
	{
		name = n;
		concentration = c;
	}

	public String getName()
	{
		return name;
	}

	public String getConcentration()
	{
		return concentration;
	}

	public void setName(String n)
	{
		name = n;
	}

	public void setConcentration(String c)
	{
		concentration = c;
	}

	public void display()
	{
		System.out.printf("%s, %s\n", getName(), getConcentration());
	}

	public static void main(String args[])
	{
	}
}

Inheritance

public class CSStudent extends Student
{
	private String concentration;//either DS or IT

	public CSStudent()
	{
	}

	public CSStudent(String n, String c)
	{
		super(n);
		concentration = c;
	}

	public String getConcentration()
	{
		return concentration;
	}

	public void setConcentration(String c)
	{
		concentration = c;
	}

	public void display()
	{
		//System.out.printf("%s, %s\n", name, concentration);//compile error, name is private field in parent, not able to be accessed directly
		System.out.printf("%s, %s\n", getName(), getConcentration());
	}

	public static void main(String args[])
	{
		CSStudent s = new CSStudent("David", "IT");
		s.display();
	}
}

  • public methods and fields will be inherited from the parent
  • private methods and fields in the parent are not inderited by child, but child can access those methods and fields by inherited methods
  • Constructor in child can use super() to call constructor in parent
