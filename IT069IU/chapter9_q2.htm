//Student.java
public class Student
{
	private String name;
	protected int UID;

	public Student()
	{
	}

	public Student(String n)
	{
		name = n;
		UID = 0;
	}

	public Student(String n, int u)
	{
		name = n;
		UID = u;
	}

	public String getName()
	{
		return name;
	}

	public int getUID()
	{
		return UID;
	}

	public void setName(String n)
	{
		name = n;
	}

	public void setUID(int u)
	{
		UID = u;
	}

	public String toString()
	{
		return name + ", " + UID;//string concatenation
	}
}
