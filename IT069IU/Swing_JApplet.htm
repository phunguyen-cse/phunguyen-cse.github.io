import java.awt.*;
import java.awt.event.*;
import javax.swing.BorderFactory;
import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JTextField;
import javax.swing.border.Border;
import javax.swing.JApplet;

public class Draw extends JApplet
{
	private JTextField text = new JTextField("32");
	private JLabel label = new JLabel("0");
	private JLabel label2 = new JLabel("°F");
	private JLabel label3 = new JLabel("°C");
	private JButton button = new JButton("Convert");
	private Border border = BorderFactory.createLineBorder(Color.BLUE, 1);

	public void init()
	{
		setSize(800, 100);
		setLayout(null);

		text.setBounds(10, 20, 200, 50);
		text.setFont(text.getFont().deriveFont(18f));
		label2.setBounds(210, 20, 50, 50);
		label2.setFont(label2.getFont().deriveFont(18f));

		label.setBorder(border);
		label.setBounds(500, 20, 200, 50);
		label.setFont(label.getFont().deriveFont(18f));
		label3.setBounds(700, 20, 50, 50);
		label3.setFont(label3.getFont().deriveFont(18f));

		button.setBounds(260, 20, 200, 50);
		button.setFont(button.getFont().deriveFont(18f));
		button.addActionListener(new ActionListener(){
			public void actionPerformed(ActionEvent e){
				double fTemperature = Double.parseDouble(text.getText());
				double cTemperature = (fTemperature-32)*5/9.0;
				String str = String.format("%.2f", cTemperature);
				label.setText(str);
			}
		});

		add(text);
		add(label2);
		add(label);
		add(label3);
		add(button);

		//setVisible(true);
	}

	/*public static void main(String args[])
	{
		//create JFrame Object as the container
		Draw draw = new Draw();
	}*/
}
