Increment and Decrement Operators
public class Assignment
{
	public static void main(String args[])
	{
		int i = 1;

		// increment operator
		System.out.println("==============Increment================");
		i++;// i = i+1
		System.out.printf("i++: %d\n", i);

		// decrement operator
		System.out.println("==============Decrement================");
		i--;// i = i-1
		System.out.printf("i--: %d\n", i);

		// implement order
		System.out.println("==============Implement Order=============");
		System.out.printf("i++: %d\n", i++);// print i, then i++
		System.out.printf("i: %d\n", i);

		System.out.printf("++i: %d\n", ++i);// i++, then print i
		System.out.printf("i: %d\n", i);
	}
}
Compound Assignment Operators
public class Compound
{
	public static void main(String args[])
	{
		int i = 10;

		// +=
		i += 1;// i = i+1
		System.out.println(i);

		// -=
		i -= 2;// i = i-2
		System.out.println(i);

		// *=
		i *= 10;
		System.out.println(i);

		// /=
		i /= 5;
		System.out.println(i);
	}
}

// Student.java
public class Student
{
	int grade;
	String name;

	// constructor
	public Student()
	{
	}

	public Student(String name, int grade)
	{
		this.grade = grade;
		this.name = name;
	}

	// accessor
	public int getGrade() {return grade;}
	public String getName() {return name;}

	public String getLetterGrade()
	{
		if (grade >= 90)
			return "A";
		else if (grade >= 80)
			return "B";
		else if (grade >= 70)
			return "C";
		else if (grade >= 60)
			return "D";
		else
			return "F";
	}

	// mutator
	public void setGrade(int grade)
	{
		this.grade = grade;
	}

	public void setName(String name)
	{
		this.name = name;
	}
}

import java.util.Scanner;

public class Average
{
	public static void main(String args[])
	{
		int grade = 0;
		int count = 0; // count the number of grades
		int sum = 0; // save the accumulation

		Scanner s = new Scanner(System.in);

		// read grades, type in a negative number to stop
		while (grade >= 0)
		{
			System.out.println("Enter the grade:");
			grade = s.nextInt();
			if (grade >= 0)
			{
				sum += grade;
				count++;
			}
		}

		System.out.printf("Average grade: %10.2f\n", (float)sum/count);
	}
}
