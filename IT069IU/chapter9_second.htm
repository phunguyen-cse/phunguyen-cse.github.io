Protected
//Student.java
public class Student
{
	private String name;
	protected int UID;

	public Student()
	{
	}

	public Student(String n)
	{
		name = n;
		UID = 0;
	}

	public Student(String n, int u)
	{
		name = n;
		UID = u;
	}

	public String getName()
	{
		return name;
	}

	public int getUID()
	{
		return UID;
	}

	public void setName(String n)
	{
		name = n;
	}

	public void setUID(int u)
	{
		UID = u;
	}
}
			
//CSStudent.java
public class CSStudent extends Student
{
	private String concentration;//either scientific or IT

	public CSStudent(String n, int u, String c)
	{
		super(n, u);
		concentration = c;
	}

	public String getConcentration()
	{
		return concentration;
	}

	public void setConcentration(String c)
	{
		concentration = c;
	}

	public void display()
	{
		//System.out.printf("%s, %s\n", name, concentration);//not able to access parent private fields directly
		System.out.printf("Name: %s\n", getName());
		System.out.printf("UID: %d\n", UID);
		System.out.printf("Concentration: %s\n", concentration);
	}
}
			
//Test.java
public class Test
{
	public static void main(String args[])
	{
		Student s = new Student("Mike", 1);
		System.out.printf("Name: %s\n", s.getName());
		System.out.printf("UID: %d\n", s.getUID());

		CSStudent c = new CSStudent("David", 2, "Scientific");
		c.display();
	}
}
  • private methods and fields in parent class are not able to be accessed directly in child class
  • protected methods and fields in parent class are inherited in child class, can be accessed directly in child class
