import java.util.Random;

public class Ran
{
	public static int getRand()
	{
		Random r = new Random();
		return r.nextInt(10);
	}

	public static void main(String args[])
	{
		int sum = 0;

		for(int i = 0; i < 5; i++)
		{
			int r = getRand();
			sum += r;
			System.out.println(r);
		}

		System.out.printf("Sum: %d\n", sum);
	}
}
