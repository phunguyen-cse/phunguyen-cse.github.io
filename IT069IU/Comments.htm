    public class Student

	private String name;

	public Student(String name, int grade)
	{
		this.grade = grade;
		this.name = name;
	}

	//accessor
	public int getGrade() {return grade;}
	public String getName() {return name;}

	public static void main(String args[])
	{
		Student s1 = new Student("David", 90);
		Student s2 = new Student("Mike", 85);
		
		System.out.printf("%s has better grade than %s\n", s1.getName(), s2.getName());
	}
}
